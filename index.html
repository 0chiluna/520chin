// --- Visualizer cạnh phải --- const canvas = document.getElementById("visualizer"); const ctx = canvas.getContext("2d"); let audioCtx, analyser, source, dataArray; function setupAudioVisualizer() { if (audioCtx) return; audioCtx = new (window.AudioContext || window.webkitAudioContext)(); analyser = audioCtx.createAnalyser(); source = audioCtx.createMediaElementSource(music); source.connect(analyser); analyser.connect(audioCtx.destination); analyser.fftSize = 128; dataArray = new Uint8Array(analyser.frequencyBinCount); drawVisualizer(); } function drawVisualizer() { requestAnimationFrame(drawVisualizer); analyser.getByteFrequencyData(dataArray); ctx.clearRect(0, 0, canvas.width, canvas.height); const barCount = 25; // số dòng const barHeight = canvas.height / barCount; for (let i = 0; i < barCount; i++) { const value = dataArray[i] / 2; const barWidth = Math.max(4, value / 6); // Gradient FFFFF0 -> FFFACD -> F0FFFF const gradient = ctx.createLinearGradient(0, 0, canvas.width, 0); gradient.addColorStop(0, "#FFFFF0"); gradient.addColorStop(0.5, "#FFFACD"); gradient.addColorStop(1, "#F0FFFF"); ctx.fillStyle = gradient; ctx.fillRect(canvas.width - barWidth, i * barHeight + 2, barWidth, barHeight - 4); } } // Khi nhạc play thì setup visualizer music.addEventListener("play", () => { setupAudioVisualizer(); if (audioCtx.state === "suspended") { audioCtx.resume(); } });
